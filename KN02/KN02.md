### Dockerfile WEB:
FROM php:8.0-apache
 
RUN docker-php-ext-install mysqli
 
COPY info.php /var/www/html/
COPY db.php /var/www/html/

EXPOSE 8080

Commands:
docker build -t kn02b-web .

docker run -d -p 80:80 --name kn02b_web --link kn02b_db kn02b-web
